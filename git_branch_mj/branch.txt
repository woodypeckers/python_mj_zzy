这是练习branch的操作
git branch mj 、 git checkout mj 效果等同于 git checkout -b mj
git checkout -b mj #本地创建并切换到mj 分支上
git branch  #查看分支
git checkout master #切换到主线master
git push origin mj #将分支发布到github上
git branch -d mj #在本地删除分支
git push origin :mj #在远端删除分支，冒号等效于删除
git push origin mj #提交到github分支上
git merge mj #首先切换到master上进行合并 git merge mj，合并之后还需要git push
